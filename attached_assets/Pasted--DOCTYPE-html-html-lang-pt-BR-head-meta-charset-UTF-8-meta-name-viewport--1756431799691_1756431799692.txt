        // SISTEMA DE LOGIN
        const ADMIN_USER = 'admin';
        const ADMIN_PASS = '123456';

        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;
            const errorDiv = document.getElementById('loginError');
            
            if (username === ADMIN_USER && password === ADMIN_PASS) {
                document.getElementById('loginScreen').style.display = 'none';
                document.getElementById('dashboard').style.display = 'block';
                loadSavedEvent();
            } else {
                errorDiv.textContent = 'Usu√°rio ou senha incorretos!';
            }
        });

        function logout() {
            document.getElementById('loginScreen').style.display = 'flex';
            document.getElementById('dashboard').style.display = 'none';
            document.getElementById('username').value = '';
            document.getElementById('password').value = '';
            document.getElementById('loginError').textContent = '';
        }

        // GERENCIAMENTO DE EVENTOS
        function saveEvent() {
            const eventData = {
                title: document.getElementById('eventTitle').value,
                subtitle: document.getElementById('eventSubtitle').value,
                description: document.getElementById('eventDescription').value,
                endDate: document.getElementById('eventEndDate').value,
                active: document.getElementById('eventActive').checked,
                timestamp: new Date().toISOString()
            };

            // Salvar no localStorage (em produ√ß√£o, salvaria no banco/API)
            localStorage.setItem('currentEvent', JSON.stringify(eventData));
            
            updateStatus();
            updatePreview();
            
            alert('‚úÖ Evento salvo com sucesso!');
        }

        function loadSavedEvent() {
            const saved = localStorage.getItem('currentEvent');
            if (saved) {
                const eventData = JSON.parse(saved);
                
                document.getElementById('eventTitle').value = eventData.title || '';
                document.getElementById('eventSubtitle').value = eventData.subtitle || '';
                document.getElementById('eventDescription').value = eventData.description || '';
                document.getElementById('eventEndDate').value = eventData.endDate || '';
                document.getElementById('eventActive').checked = eventData.active || false;
                
                updateStatus();
                updatePreview();
            }
        }

        function clearEvent() {
            document.getElementById('eventTitle').value = '';
            document.getElementById('eventSubtitle').value = '';
            document.getElementById('eventDescription').value = '';
            document.getElementById('eventEndDate').value = '';
            document.getElementById('eventActive').checked = false;
            
            localStorage.removeItem('currentEvent');
            
            updateStatus();
            document.getElementById('previewBox').innerHTML = `
                <div style="text-align: center; color: #999; padding: 40px;">
                    <h3>Preview aparecer√° aqui</h3>
                    <p>Configure um evento para ver como ficar√° no site</p>
                </div>
            `;
        }

        function updateStatus() {
            const active = document.getElementById('eventActive').checked;
            const title = document.getElementById('eventTitle').value;
            const indicator = document.getElementById('statusIndicator');
            const text = document.getElementById('statusText');
            
            if (active && title) {
                indicator.classList.add('active');
                text.textContent = `Evento ativo: ${title}`;
            } else {
                indicator.classList.remove('active');
                text.textContent = 'Nenhum evento ativo';
            }
        }

        function updatePreview() {
            const title = document.getElementById('eventTitle').value;
            const subtitle = document.getElementById('eventSubtitle').value;
            const description = document.getElementById('eventDescription').value;
            const active = document.getElementById('eventActive').checked;
            
            if (active && title) {
                document.getElementById('previewBox').innerHTML = `
                    <div class="event-banner">
                        <h3>${title}</h3>
                        <p>${subtitle}</p>
                        <p style="font-size: 14px; opacity: 0.9;">${description}</p>
                        <a href="#" class="cta">CADASTRE-SE AGORA</a>

                        <h3>Evento desativado</h3>
                        <p>Ative o evento para ver o preview</p>
                    </div>
                `;
            }
        }
