    <script>
        // Adicionar mais partículas dinamicamente
        function createDynamicParticles() {
            const container = document.querySelector('.logo-container');
            const codeSymbols = ['&lt;/&gt;', '{}', '[]', '()', 'def', 'py', '$', '#', '&amp;&amp;', '||'];
            
            setInterval(() => {
                const particle = document.createElement('div');
                particle.className = 'code-particle';
                particle.textContent = codeSymbols[Math.floor(Math.random() * codeSymbols.length)];
                particle.style.position = 'absolute';
                particle.style.left = Math.random() * 100 + '%';
                particle.style.top = Math.random() * 100 + '%';
                particle.style.animation = `particle-drift ${Math.random() * 4 + 4}s linear`;
                particle.style.animationDelay = '0s';
                particle.style.color = Math.random() > 0.5 ? '#00d4aa' : '#f39c12';
                
                container.appendChild(particle);
                
                setTimeout(() => {
                    particle.remove();
                }, 8000);
            }, 2000);
        }

        // Efeito de mordida da cobra no hover
        const snake = document.querySelector('.snake');
        const mainCircle = document.querySelector('.main-circle');
        
        mainCircle.addEventListener('mouseenter', () => {
            snake.style.transform = 'scale(1.1) rotate(5deg)';
            snake.style.filter = 'brightness(1.2)';
        });
        
        mainCircle.addEventListener('mouseleave', () => {
            snake.style.transform = 'scale(1)';
            snake.style.filter = 'none';
        });

        // Efeito de piscar dos olhos da cobra
        function blinkSnakeEyes() {
            const head = document.querySelector('.snake-head');
            setInterval(() => {
                head.style.opacity = '0.3';
                setTimeout(() => {
                    head.style.opacity = '1';
                }, 150);
            }, 3000);
        }

        // Sincronizar animação das letras com a respiração da cobra
        function syncAnimations() {
            const letters = document.querySelectorAll('.letter-orbit');
            const snake = document.querySelector('.snake');
            
            setInterval(() => {
                letters.forEach((letter, index) => {
                    letter.style.textShadow = `0 0 ${30 + Math.sin(Date.now() * 0.003 + index) * 10}px currentColor`;
                });
            }, 50);
        }

        // Inicializar efeitos
        createDynamicParticles();
        blinkSnakeEyes();
        syncAnimations();

        // Efeito especial no clique
        document.addEventListener('click', (e) => {
            const ripple = document.createElement('div');
            ripple.style.position = 'absolute';
            ripple.style.left = e.clientX + 'px';
            ripple.style.top = e.clientY + 'px';
            ripple.style.width = '100px';
            ripple.style.height = '100px';
            ripple.style.background = 'radial-gradient(circle, rgba(241,196,15,0.4), transparent)';
            ripple.style.borderRadius = '50%';
            ripple.style.transform = 'translate(-50%, -50%)';
            ripple.style.animation = 'ripple-effect 1s ease-out';
            ripple.style.pointerEvents = 'none';
            ripple.style.zIndex = '1000';
            
            document.body.appendChild(ripple);
            
            setTimeout(() => {
                ripple.remove();
            }, 1000);
        });

        // Adicionar animação de ripple
        const style = document.createElement('style');
        style.textContent = `
            @keyframes ripple-effect {
                0% { transform: translate(-50%, -50%) scale(0); opacity: 1; }
                100% { transform: translate(-50%, -50%) scale(4); opacity: 0; }
            }
        `;
        document.head.appendChild(style);
    </script>
