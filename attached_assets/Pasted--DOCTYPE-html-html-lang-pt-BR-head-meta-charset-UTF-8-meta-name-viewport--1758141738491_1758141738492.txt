        let currentProject = '';
        let currentImageIndex = 0;

        function openModal(projectKey, imageIndex) {
            const modal = document.getElementById('imageModal');
            const modalImage = document.getElementById('modalImage');
            const modalTitle = document.getElementById('modalTitle');
            const modalDescription = document.getElementById('modalDescription');
            const imageCounter = document.getElementById('imageCounter');
            const thumbnails = document.getElementById('thumbnails');

            currentProject = projectKey;
            currentImageIndex = imageIndex;

            const project = projectData[projectKey];
            
            modalTitle.textContent = project.title;
            modalImage.src = project.images[imageIndex].src;
            modalDescription.textContent = project.images[imageIndex].description;
            imageCounter.textContent = `${imageIndex + 1} / ${project.images.length}`;

            // Criar thumbnails
            thumbnails.innerHTML = '';
            project.images.forEach((img, index) => {
                const thumbnail = document.createElement('img');
                thumbnail.src = img.src;
                thumbnail.className = 'thumbnail' + (index === imageIndex ? ' active' : '');
                thumbnail.onclick = () => goToImage(index);
                thumbnails.appendChild(thumbnail);
            });

            modal.classList.add('active');
            document.body.style.overflow = 'hidden';
        }

        function closeModal() {
            const modal = document.getElementById('imageModal');
            modal.classList.remove('active');
            document.body.style.overflow = 'auto';
        }

        function changeImage(direction) {
            const project = projectData[currentProject];
            currentImageIndex += direction;
            
            if (currentImageIndex >= project.images.length) {
                currentImageIndex = 0;
            } else if (currentImageIndex < 0) {
                currentImageIndex = project.images.length - 1;
            }
            
            updateModalImage();
        }

        function goToImage(index) {
            currentImageIndex = index;
            updateModalImage();
        }

        function updateModalImage() {
            const modalImage = document.getElementById('modalImage');
            const modalDescription = document.getElementById('modalDescription');
            const imageCounter = document.getElementById('imageCounter');
            const thumbnails = document.querySelectorAll('.thumbnail');

            const project = projectData[currentProject];
            
            modalImage.src = project.images[currentImageIndex].src;
            modalDescription.textContent = project.images[currentImageIndex].description;
            imageCounter.textContent = `${currentImageIndex + 1} / ${project.images.length}`;

            // Atualizar thumbnails ativas
            thumbnails.forEach((thumb, index) => {
                thumb.classList.toggle('active', index === currentImageIndex);
            });
        }

        // Fechar modal com ESC
        document.addEventListener('keydown', function(event) {
            if (event.key === 'Escape') {
                closeModal();
            } else if (event.key === 'ArrowLeft') {
                changeImage(-1);
            } else if (event.key === 'ArrowRight') {
                changeImage(1);
            }
        });

        // Fechar modal clicando fora da imagem
        document.getElementById('imageModal').addEventListener('click', function(event) {
            if (event.target === this) {
                closeModal();
            }
        });
